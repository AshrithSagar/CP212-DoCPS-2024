\newcommand{\GetBit}[2]{%
    \edef\BitPos{#2}%
    \StrLen{#1}[\StringLen]%
    \StrChar{#1}{#2}[\Bit]%
}
\newcommand{\SetBits}[2]{%
    \foreach \i in {1,...,#2} {%
            \GetBit{#1}{\i}%
            \node at (\i - 0.5, 0.5) {\LARGE \textbf{\Bit}};
        }%
}
\newcommand{\drawBraceLabel}[4]{%
    \draw [decorate, decoration={brace, amplitude=5pt}] (#1, -0.1) -- (#2, -0.1);
    \node [below] at (#3, -0.3) {\parbox{3cm}{\centering\texttt{#4}}};
}
